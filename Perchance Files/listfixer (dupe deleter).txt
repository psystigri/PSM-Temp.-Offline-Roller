$output(arr) =>
  // Use a Set to remove duplicates
  const uniqueSet = new Set(arr);

  // Convert the Set back to an array
  const uniqueArray = Array.from(uniqueSet);

  // Remove instances of "Fusion"
  const resultArray = uniqueArray.filter(element => element !== "Fusion" && element !== "Basic" && element !== "Ditto" && element !== "noDelta");

  return resultArray;
