//----------------------------------- SPLIT EVO BULLSHIT ---------------------------------------

speciesFixer(arr, specieExceptions) =>
  const exceptions = specieExceptions.split(" ** ");
  // Create a map to count occurrences of items in the exceptions list
  const exceptionsCount = new Map();

  // Iterate over the array to count exceptions and create a unique set for other items
  const uniqueSet = new Set();
  arr.forEach(item => {
    if (exceptions.includes(item)) {
      exceptionsCount.set(item, (exceptionsCount.get(item) || 0) + 1);
    } else if (item !== "Fusion" && item !== "Basic" && item !== "Ditto" && item !== "noDelta") {
      uniqueSet.add(item);
    }
  });

  // Convert the unique set back to an array
  const uniqueArray = Array.from(uniqueSet);

  // Add the counted exceptions back to the array, ensuring no more than two of each
  exceptionsCount.forEach((count, item) => {
    const limitedCount = Math.min(count, 2);
    for (let i = 0; i < limitedCount; i++) {
      uniqueArray.push(item);
    }
  });

  return uniqueArray;

extractDelta(inputText) =>
  const regex = /Delta\(([^)]+)\)/;
  const match = inputText.match(regex);
  
  return match ? match[1] : "noDelta";

removeDelta(inputText) =>
  const regex = /Delta\([^)]+\)/;
  return inputText.replace(regex, 'Delta');
